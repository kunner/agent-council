# 01. 프로젝트 셋업 가이드

Agent Council 모노레포를 처음부터 구성하는 단계별 가이드입니다.
`packages/server` (Node.js + Express + Socket.IO), `packages/web` (Vite + React + TypeScript), `packages/shared` (공유 타입/유틸) 세 패키지로 구성됩니다.

---

## 1. 사전 요구사항

아래 도구들이 모두 설치되어 있어야 합니다.

| 도구 | 최소 버전 | 설치 확인 |
|---|---|---|
| Node.js | 20.x LTS 이상 | `node -v` |
| pnpm | 9.x 이상 | `pnpm -v` |
| Claude Code CLI | 최신 | `claude -v` |
| Git | 2.38 이상 (worktree 지원) | `git --version` |
| Firebase CLI | 13.x 이상 | `firebase --version` |

### 도구 설치

```bash
# Node.js 20 LTS (nvm 사용 권장)
nvm install 20
nvm use 20

# pnpm
npm install -g pnpm

# Claude Code CLI
npm install -g @anthropic-ai/claude-code

# Firebase CLI
npm install -g firebase-tools
firebase login

# Firebase CLI 로그인 확인
firebase projects:list
```

/** Set 인덱스로 색상 반환 (순환) */

    medium: 2,
    low: 1,
  }
  return weights[priority]

/** 태스크 상태가 완료 단계인지 확인 */
export function isTerminalStatus(status: TaskStatus): boolean {

}

/** 브랜치명 생성: set-{id}/{topic} 형식 */
export function buildBranchName(setId: string, topic: string): string {
  const slug = topic
    .toLowerCase()
    .replace(/[^a-z0-9가-힣]+/g, '-')
    .replace(/^-|-$/g, '')
  return `set-${setId}/${slug}`
}

/** worktree 경로 생성 */
export function buildWorktreePath(
  basePath: string,
  projectId: string,
  setId: string
): string {
  return `${basePath}/${projectId}/set-${setId}`
}

/** ISO 8601 날짜를 사람이 읽기 편한 형식으로 변환 */
export function formatRelativeTime(date: Date): string {
  const diff = Date.now() - date.getTime()
  const minutes = Math.floor(diff / 60_000)
  if (minutes < 1) return '방금 전'
  if (minutes < 60) return `${minutes}분 전`
  const hours = Math.floor(minutes / 60)
  if (hours < 24) return `${hours}시간 전`
  const days = Math.floor(hours / 24)
  return `${days}일 전`
}

### 4.3 Express + Socket.IO 초기화

`packages/server/src/index.ts`:

```typescript
import 'dotenv/config'
import { createServer } from 'node:http'
import express from 'express'
import cors from 'cors'
import { Server as SocketIOServer } from 'socket.io'
import { initFirebase } from './firebase/admin.js'
import { registerRoutes } from './routes.js'
import { registerSocketHandlers } from './ws/handlers.js'

const PORT = Number(process.env.PORT ?? 3001)
const CLIENT_URL = process.env.CLIENT_URL ?? 'http://localhost:5173'

// Firebase Admin 초기화 (가장 먼저)
await initFirebase()

const app = express()
const httpServer = createServer(app)

// ─── 미들웨어 ──────────────────────────────────────────────
app.use(cors({ origin: CLIENT_URL, credentials: true }))
app.use(express.json({ limit: '1mb' }))

// ─── Socket.IO ─────────────────────────────────────────────
const io = new SocketIOServer(httpServer, {
  cors: { origin: CLIENT_URL, methods: ['GET', 'POST'] },
  path: '/ws',
})

registerSocketHandlers(io)

// ─── REST API 라우트 ───────────────────────────────────────
registerRoutes(app)

// ─── 헬스체크 ──────────────────────────────────────────────
app.get('/health', (_req, res) => {
  res.json({ status: 'ok', timestamp: new Date().toISOString() })
})

httpServer.listen(PORT, () => {
  console.log(`Council Server running on http://localhost:${PORT}`)
})

export { io }
```

### 4.4 Firebase Admin SDK 설정

`packages/server/src/firebase/admin.ts`:

```typescript
import admin from 'firebase-admin'

let initialized = false

export async function initFirebase(): Promise<void> {
  if (initialized) return

  const serviceAccountPath = process.env.FIREBASE_SERVICE_ACCOUNT_PATH
  const projectId = process.env.FIREBASE_PROJECT_ID

  if (!projectId) {
    throw new Error('FIREBASE_PROJECT_ID 환경변수가 설정되지 않았습니다.')
  }

  if (serviceAccountPath) {
    // 서비스 계정 JSON 파일 경로로 초기화 (로컬 개발 / 서버)
    const serviceAccount = await import(serviceAccountPath, {
      assert: { type: 'json' },
    })
    admin.initializeApp({
      credential: admin.credential.cert(serviceAccount.default),
      projectId,
    })
  } else {
    // Application Default Credentials (Google Cloud 환경)
    admin.initializeApp({
      credential: admin.credential.applicationDefault(),
      projectId,
    })
  }

  initialized = true
  console.log(`Firebase Admin initialized (project: ${projectId})`)
}

export function getFirestore(): admin.firestore.Firestore {
  return admin.firestore()
}

export function getAuth(): admin.auth.Auth {
  return admin.auth()
}
```

> **주의**: `initFirebase`는 `async function`으로 선언되어야 합니다 (`await import()` 사용). `index.ts`의 최상위에서 `await initFirebase()`를 호출하려면 ES 모듈 top-level await를 활용하거나, `async` IIFE로 감싸야 합니다. 실제 구현에서는 `fs.readFileSync` + `JSON.parse`를 사용하거나 환경변수로 서비스 계정 JSON 문자열을 주입하는 방식이 더 안전합니다.

`packages/server/src/ws/handlers.ts`:

```typescript
import type { Server } from 'socket.io'
import type { WsEvent } from '@agent-council/shared'

export function registerSocketHandlers(io: Server): void {
  io.on('connection', (socket) => {
    console.log(`Client connected: ${socket.id}`)

    socket.on('join_project', (projectId: string) => {
      void socket.join(`project:${projectId}`)
      console.log(`Socket ${socket.id} joined project:${projectId}`)
    })

    socket.on('leave_project', (projectId: string) => {
      void socket.leave(`project:${projectId}`)
    })

    socket.on('disconnect', () => {
      console.log(`Client disconnected: ${socket.id}`)
    })
  })
}

/** 프로젝트 방에 WsEvent 브로드캐스트 (서버 내부용) */
export function broadcastToProject(
  io: Server,
  projectId: string,
  event: WsEvent
): void {
  io.to(`project:${projectId}`).emit('ws_event', event)
}
```

### 5.6 React Router 설정

`packages/web/src/App.tsx`:

```tsx
import { useEffect } from 'react'
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom'
import { onAuthStateChanged } from 'firebase/auth'
import { auth } from './firebase/config.js'
import { useAuthStore } from './stores/index.js'
import LoginPage from './pages/LoginPage.js'
import DashboardPage from './pages/DashboardPage.js'
import NewProjectPage from './pages/NewProjectPage.js'
import CouncilRoomPage from './pages/CouncilRoomPage.js'
import SettingsPage from './pages/SettingsPage.js'

function RequireAuth({ children }: { children: React.ReactNode }) {
  const { user, loading } = useAuthStore()
  if (loading) return <div className="flex h-screen items-center justify-center">Loading...</div>
  if (!user) return <Navigate to="/login" replace />
  return <>{children}</>
}

export default function App() {
  const { setUser, setLoading } = useAuthStore()

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, (user) => {
      setUser(user)
      setLoading(false)
    })
    return unsubscribe
  }, [setUser, setLoading])

  return (
    <BrowserRouter>
      <Routes>
        <Route path="/login" element={<LoginPage />} />
        <Route
          path="/"
          element={
            <RequireAuth>
              <DashboardPage />
            </RequireAuth>
          }
        />
        <Route
          path="/new"
          element={
            <RequireAuth>
              <NewProjectPage />
            </RequireAuth>
          }
        />
        <Route
          path="/p/:projectId"
          element={
            <RequireAuth>
              <CouncilRoomPage />
            </RequireAuth>
          }
        />
        <Route
          path="/settings"
          element={
            <RequireAuth>
              <SettingsPage />
            </RequireAuth>
          }
        />
        <Route path="*" element={<Navigate to="/" replace />} />
      </Routes>
    </BrowserRouter>
  )
}
```

### 6.7 Firestore Security Rules

`firebase/firestore.rules`:

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 사용자는 자신의 문서만 읽기/쓰기
    match /users/{userId} {
      allow read, write: if request.auth != null
        && request.auth.uid == userId;
    }

    // 프로젝트: 소유자만 접근
    match /projects/{projectId} {
      allow read, write: if request.auth != null
        && resource.data.ownerId == request.auth.uid;

      // 새 프로젝트 생성: 본인이 ownerId로 설정해야 함
      allow create: if request.auth != null
        && request.resource.data.ownerId == request.auth.uid;

      // Git 설정
      match /git/{docId} {
        allow read, write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;
      }

      // Council Room + 메시지
      match /rooms/{roomId} {
        allow read, write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;

        match /messages/{messageId} {
          allow read: if request.auth != null;
          allow create: if request.auth != null;
        }
      }

      // Agent Set + 내부 로그
      match /sets/{setId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;

        match /logs/{logId} {
          allow read: if request.auth != null;
          allow create: if request.auth != null;
        }
      }

      // 태스크 보드
      match /tasks/{taskId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;
      }

      // PR 추적
      match /pullRequests/{prId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;
      }

      // 프로젝트 스냅샷
      match /snapshots/{snapshotId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null
          && get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId
             == request.auth.uid;
      }
    }

    // 토큰 사용량: 본인만 읽기/쓰기
    match /usage/{userId}/{document=**} {
      allow read, write: if request.auth != null
        && request.auth.uid == userId;
    }
  }
}
```
